PUNTO 2 a)

SINTAXIS
    GRUPOVACIO -> GRUPO
    AGREGAR: GRUPO X ELEMENTO -> GRUPO
    ESVACIO: GRUPO -> BOOL
    CANTIDAD: GRUPO -> ENTERO >= 0
    QUITARULTIMO: GRUPO -> GRUPO
    ULTIMOELEMENTO: GRUPO -> ELEMENTO U {indefinido}
    INCLUIDO: GRUPO X ELEMENTO -> BOOL
    CONTARCOMUNES: GRUPO X GRUPO -> ENTERO >= 0

SEMANTICA: ∀ C, T ∈ GRUPO, ∀ x,b ∈ ENTERO
    ESVACIO(GRUPOVACIO) = true
    ESVACIO(AGREGAR(C, x)) = false
    CANTIDAD(GRUPOVACIO) = 0
    CANTIDAD(AGREGAR(C, x)) = 1 + CANTIDAD(C)
    QUITARULTIMO(GRUPOVACIO) = GRUPOVACIO
    QUITARULTIMO(AGREGAR(C, x)) = C
    ULTIMOELEMENTO(GRUPOVACIO) = indefinido
    ULTIMOELEMENTO(AGREGAR(C, x)) = x
    INCLUIDO(GRUPOVACIO,x) = false
    INCLUIDO(AGREGAR(C,x),b) = SI x = b ENTONCES true SINO INCLUIDO(C,b)
    CONTARCOMUNES(GRUPOVACIO,GRUPOVACIO) = 0
    CONTARCOMUNES(AGREGAR(C, x),GRUPOVACIO) = 0
    CONTARCOMUNES(GRUPOVACIO,AGREGAR(T,b)) = 0
    CONTARCOMUNES(AGREGAR(C,x),AGREGAR(T,b)) = ...
        SI INCLUIDO(AGREGAR(C,x), b) ENTONCES 
            1 + CONTARCOMUNES(AGREGAR(C,x), T)
        SINO 
            CONTARCOMUNES(AGREGAR(C,x), T)

    SI ESVACIO(T)        
        RETORNA C
    SINO
        RETORNA AGREGAR( union( C, QUITARULTIMO(T) ), ULTIMOELEMENTO(T) )
