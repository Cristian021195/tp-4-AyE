TIPO GRUPO(ELEMENTO)
OPERACIONES
    
SINTAXIS:
    GRUPOVACIO: -> GRUPO
    AGREGAR: GRUPO x ELEMENTO -> GRUPO
    ESVACIO: GRUPO -> BOOL
    CANTIDAD: GRUPO -> entero >= 0
    QUITARULTIMO: GRUPO -> GRUPO
    ULTIMOELEMENTO: GRUPO -> ELEMENTO | indef
    INCLUIDO: GRUPO x ELEMENTO -> BOOL
    CONTARCOMUNES: GRUPO x GRUPO -> BOOL
    CONTARCOMUNES_ALT: GRUPO x GRUPO -> BOOL

SEMANTICA: para todo G1,G2,G3 de GRUPO, para todo a,b,c de ELEMENTO
    ESVACIO(GRUPOVACIO) = TRUE
    ESVACIO(AGREGAR(G1, a)) = FALSE
    CANTIDAD(GRUPOVACIO) = 0
    CANTIDAD(AGREGAR(G1, a)) = 1 + CANTIDAD(G1)
    QUITARULTIMO(GRUPOVACIO) = indef
    QUITARULTIMO(AGREGAR(G1, a)) = a
    INCLUIDO(GRUPOVACIO, a) = FALSE
    INCLUIDO(AGREGAR(G1, a), b) = ...
        SI(a == b) ENT TRUE
        SINO INCLUIDO(G1, b)
    CONTARCOMUNES(G1,G2) = ...
        SI( ESVACIO(G1) OR ESVACIO(G2) ) ENT 0
        SINO
            SI( ULTIMOELEMENTO(G1) == ULTIMOELEMENTO(G2) ) ENT
                1 + CONTARCOMUNES(QUITARULTIMO(G1), QUITARULTIMO(G2))
            SINO
                CONTARCOMUNES(G1, AGREGAR(QUITARULTIMO(G2), ULTIMO(G2)))

    CONTARCOMUNES_ALT(G1, G2) = ...
        SI( ESVACIO(G1) OR ESVACIO(G2) ) ENT 0
        SINO
            SI ( INCLUIDO(G2, ULTIMOELEMENTO(G1)) )
                1 + CONTARCOMUNES_ALT( QUITARULTIMO(G1), G2)
            SINO
                CONTARCOMUNES_ALT( QUITARULTIMO(G1), G2)
